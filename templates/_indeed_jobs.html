_indeed_jobs

Job listings scraped from Indeed. Records include job title, company name, location, raw pay string, normalized min/max hourly rate, employment type, schedule tags, benefits, and whether Easy Apply is available.

Example Request
Python · requests 
import requests

# Replace these with the credentials provided by your API platform
# (e.g. RapidAPI, your own deployment, etc.)
API_BASE_URL = "https://<your-api-host>"
API_KEY      = "<your-api-key>"

url = f"{API_BASE_URL}/v1/collections/_indeed_jobs/search"

headers = {
    "Content-Type":  "application/json",
    "X-Api-Key":     API_KEY,  # header name may vary by platform
}

# Structure the POST body however you need — all fields are optional.
# Use "filter" to narrow results, "order_by" to sort, "page_number" to paginate.
payload = {
    "select":      ["*"],
    "filter":      {
        "country": "United States",
        "is_easy_apply": {"$gte": 0, "$lte": 100},
        "title": {"$like": "%keyword%"},
        "city": {"$in": ["New York", "Los Angeles"]},
    },
    "order_by":    [{"field": "pay_min", "direction": "asc"}],
    "page_number": 1,
}

resp = requests.post(url, json=payload, headers=headers)
print(resp.status_code)
print(resp.json())
