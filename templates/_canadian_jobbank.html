_canadian_jobbank

Job listings scraped from the Canadian Job Bank. Records include job title, company name, location (city/province), posted date, and salary. All salary data is normalized to an hourly rate. Includes tags for application types (Direct Apply, LMIA requested, etc.).

import requests

# Replace these with the credentials provided by your API platform
# (e.g. RapidAPI, your own deployment, etc.)
API_BASE_URL = "https://<your-api-host>"
API_KEY      = "<your-api-key>"

url = f"{API_BASE_URL}/v1/collections/_canadian_jobbank/search"

headers = {
    "Content-Type":  "application/json",
    "X-Api-Key":     API_KEY,  # header name may vary by platform
}

# Structure the POST body however you need — all fields are optional.
# Use "filter" to narrow results, "order_by" to sort, "page_number" to paginate.
payload = {
    "select":      ["*"],
    "filter":      {
        "country": "United States",
        "is_lmia": {"$gte": 0, "$lte": 100},
        "is_direct_apply": {"$gte": 0},
        "title": {"$like": "%keyword%"},
        "state": {"$in": ["Texas", "California"]},
    },
    "order_by":    [{"field": "posted_date", "direction": "asc"}],
    "page_number": 1,
}

resp = requests.post(url, json=payload, headers=headers)
print(resp.status_code)
print(resp.json())
