Filter Operators
Operator Meaning Example
$gte Greater than or equal to {"price": {"$gte": 5000}}
$lte Less than or equal to {"price": {"$lte": 20000}}
$gt Strictly greater than {"year": {"$gt": 2010}}
$lt Strictly less than {"mileage": {"$lt": 100000}}
$eq Exact match (shorthand: pass the value directly) {"country": "United States"}
$ne Not equal to {"country": {"$ne": "Canada"}}
$like SQL LIKE pattern match — use % as wildcard {"title": {"$like": "%toyota%"}}
$in Value must be in the provided list {"state": {"$in": ["Texas", "California"]}}
Example Request
Python · requests 
import requests

# Replace these with the credentials provided by your API platform
# (e.g. RapidAPI, your own deployment, etc.)
API_BASE_URL = "https://<your-api-host>"
API_KEY      = "<your-api-key>"

url = f"{API_BASE_URL}/v1/collections/_craigslist_cars/search"

headers = {
    "Content-Type":  "application/json",
    "X-Api-Key":     API_KEY,  # header name may vary by platform
}

# Structure the POST body however you need — all fields are optional.
# Use "filter" to narrow results, "order_by" to sort, "page_number" to paginate.
payload = {
    "select":      ["*"],
    "filter":      {
        "country":  "United States",
        "price":    {"$gte": 1000, "$lte": 15000},
        "year":     {"$gte": 2010},
        "mileage":  {"$lt":  150000},
        "title":    {"$like": "%ford%"},
    },
    "order_by":    [{"field": "price", "direction": "asc"}],
    "page_number": 1,
}

resp = requests.post(url, json=payload, headers=headers)
print(resp.status_code)
print(resp.json())
